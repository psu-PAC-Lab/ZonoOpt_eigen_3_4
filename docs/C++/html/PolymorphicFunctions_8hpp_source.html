198inline std::unique_ptr<HybZono> intersection(const HybZono& Z1, HybZono& Z2, const Eigen::SparseMatrix<zono_float>& R)
199{
200 // handle default arguments
201 const Eigen::SparseMatrix<zono_float> * R_ptr = nullptr;
202 Eigen::SparseMatrix<zono_float> R_def;
203 if (R.rows() == 0 && R.cols() == 0)
204 {
205 R_def.resize(Z1.n, Z1.n);
206 R_def.setIdentity();
207 R_ptr = &R_def;
208 }
209 else
210 {
211 R_ptr = &R;
212 }
213
214 // check dimensions
215 if (R_ptr->rows() != Z2.n || R_ptr->cols() != Z1.n)
216 {
217 throw std::invalid_argument("Intersection: inconsistent input dimensions.");
218 }
219
220 // early exit
221 if (Z1.is_empty_set() || Z2.is_empty_set()) {
222 return std::make_unique<EmptySet>(Z1.n);
223 }
224
225 // make sure Z1 and Z2 both using same generator range
226 if (Z1.zero_one_form != Z2.zero_one_form)
227 {
228 Z2.convert_form();
229 }
230
231 // compute intersection
232 Eigen::SparseMatrix<zono_float> Gc = Z1.Gc;
233 Gc.conservativeResize(Z1.n, Z1.nGc + Z2.nGc);
234
235 Eigen::SparseMatrix<zono_float> Gb = Z1.Gb;
236 Gb.conservativeResize(Z1.n, Z1.nGb + Z2.nGb);
237
238 Eigen::Vector<zono_float, -1> c = Z1.c;
239
240 std::vector<Eigen::Triplet<zono_float>> tripvec;
241 Eigen::SparseMatrix<zono_float> Ac (Z1.nC + Z2.nC + R_ptr->rows(), Z1.nGc + Z2.nGc);
242 get_triplets_offset<zono_float>(Z1.Ac, tripvec, 0, 0);
243 get_triplets_offset<zono_float>(Z2.Ac, tripvec, Z1.nC, Z1.nGc);
244 Eigen::SparseMatrix<zono_float> RZ1Gc = (*R_ptr)*Z1.Gc;
245 get_triplets_offset<zono_float>(RZ1Gc, tripvec, Z1.nC + Z2.nC, 0);
246 Eigen::SparseMatrix<zono_float> mZ2Gc = -Z2.Gc;
247 get_triplets_offset<zono_float>(mZ2Gc, tripvec, Z1.nC + Z2.nC, Z1.nGc);
248 Ac.setFromTriplets(tripvec.begin(), tripvec.end());
249
250 tripvec.clear();
251 Eigen::SparseMatrix<zono_float> Ab (Z1.nC + Z2.nC + R_ptr->rows(), Z1.nGb + Z2.nGb);
252 get_triplets_offset<zono_float>(Z1.Ab, tripvec, 0, 0);
253 get_triplets_offset<zono_float>(Z2.Ab, tripvec, Z1.nC, Z1.nGb);
254 Eigen::SparseMatrix<zono_float> RZ1Gb = (*R_ptr)*Z1.Gb;
255 get_triplets_offset<zono_float>(RZ1Gb, tripvec, Z1.nC + Z2.nC, 0);
256 Eigen::SparseMatrix<zono_float> mZ2Gb = -Z2.Gb;
257 get_triplets_offset<zono_float>(mZ2Gb, tripvec, Z1.nC + Z2.nC, Z1.nGb);
258 Ab.setFromTriplets(tripvec.begin(), tripvec.end());
259
260
261 Eigen::Vector<zono_float, -1> b (Z1.nC + Z2.nC + R_ptr->rows());
262 b.segment(0, Z1.nC) = Z1.b;
263 b.segment(Z1.nC, Z2.nC) = Z2.b;
264 b.segment(Z1.nC + Z2.nC, R_ptr->rows()) = Z2.c - (*R_ptr)*Z1.c;
265
266 // return correct output type
267 if (Z1.is_hybzono() || Z2.is_hybzono())
268 return std::make_unique<HybZono>(Gc, Gb, c, Ac, Ab, b, Z1.zero_one_form);
269 else
270 return std::make_unique<ConZono>(Gc, c, Ac, b, Z1.zero_one_form);
271}

304inline std::unique_ptr<HybZono> halfspace_intersection(HybZono& Z, const Eigen::SparseMatrix<zono_float>& H,
305 const Eigen::Vector<zono_float, -1>& f, const Eigen::SparseMatrix<zono_float>& R)
306{
307 // use the constrain function
308
309 // convert H to row-major to efficiently convert into Inequality form
310 Eigen::SparseMatrix<zono_float, Eigen::RowMajor> H_rm = H;
311 std::vector<Inequality> ineqs;
312 for (int k=0; k<H_rm.rows(); ++k)
313 {
314 // get row of constraint matrix
315 std::vector<Eigen::Triplet<zono_float>> trips_row = get_triplets_row<zono_float>(H_rm, k);
316
317 // build constraint
318 Inequality ineq(static_cast<int>(H_rm.cols()));
319 for (const auto& trip : trips_row)
320 {
321 ineq.add_term(trip.col(), trip.value());
322 }
323 ineq.set_rhs(f(k));
324 ineq.set_ineq_type(LESS_OR_EQUAL);
325
326 ineqs.push_back(ineq);
327 }
328
329 // call constrain
330 return constrain(Z, ineqs, R);

334inline std::unique_ptr<HybZono> pontry_diff(HybZono& Z1, HybZono& Z2, bool exact)
335{
336 // check dimensions
337 if (Z1.n != Z2.n)
338 {
339 throw std::invalid_argument("Pontryagin difference: n dimensions must match.");
340 }
341
342 // check empty set inputs
343 if (Z1.is_empty_set() || Z2.is_empty_set())
344 return std::unique_ptr<HybZono>(Z1.clone());
345
346 // check point inputs
347 if (Z1.nG == 0 && !exact)
348 return std::make_unique<EmptySet>(Z1.n);
349 if (Z2.nG == 0)
350 exact = true; // easy to compute exactly
351
352 // check no exact pontry diff with conzono subtrahend
353 if (exact && (Z2.is_conzono() || Z2.is_hybzono()))
354 throw std::invalid_argument("Pontryagin difference: cannot compute exact set when subtrahend is a ConZono or HybZono");
355
356 // require Z1 and Z2 to be in [-1,1] form
357 if (Z1.zero_one_form) Z1.convert_form();
358 if (Z2.zero_one_form) Z2.convert_form();
359
360 // exact case
361 if (exact)
362 {
363 // init Zout
364 std::unique_ptr<HybZono> Z_out (Z1.clone());
365 Z_out->c -= Z2.c;
366
367 // iteratively compute pontryagin difference from columns of Z2 generator matrix
368 const Eigen::Matrix<zono_float, -1, -1> G2 = Z2.G.toDense();
369
370 for (int i=0; i<Z2.nG; ++i)
371 {
372 std::unique_ptr<HybZono> Z_plus (Z_out->clone()), Z_minus (Z_out->clone());
373 Z_plus->c += G2.col(i);
374 Z_minus->c -= G2.col(i);
375 Z_out = intersection(*Z_plus, *Z_minus);
376 }
377
378 return Z_out;
379 }
380
381 // inexact case
382
383 // logic to handle hybrid zonotope cases. Avoiding recursion to prevent redundant get_leaves calculations.
384 if (Z1.is_hybzono() && Z2.is_hybzono())
385 {
386 // get leaves and convert to zonos
387 auto Z1_leaves = Z1.get_leaves();
388 auto Z2_CZ_leaves = Z2.get_leaves();
389 std::vector<Zono> Z2_leaves;
390 for (auto& CZ : Z2_CZ_leaves)
391 {
392 Z2_leaves.push_back(*CZ.to_zono_approx());
393 }
394
395 // take union of pontry diffs for inner approx
396 std::vector<std::shared_ptr<HybZono>> leaf_diffs; // init
397 for (auto& Z1_leaf : Z1_leaves)
398 {
399 std::unique_ptr<HybZono> Z_out; // declare
400 for (auto& Z2_leaf : Z2_leaves)
401 {
402 if (!Z_out) {
403 Z_out = pontry_diff(Z1_leaf, Z2_leaf, false);
404 }
405 else {
406 Z_out = intersection(*Z_out, *pontry_diff(Z1_leaf, Z2_leaf, false));
407 }
408 }
409 leaf_diffs.push_back(std::move(Z_out));
410 }
411 return union_of_many(leaf_diffs);
412 }
413 else if (Z1.is_hybzono())
414 {
415 // get leaves and convert to zonos
416 auto Z1_leaves = Z1.get_leaves();
417 auto Z2_CZ = dynamic_cast<const ConZono*>(&Z2);
418 Zono Z2_zono = *Z2_CZ->to_zono_approx();
419
420 // take union of pontry diffs for inner approx
421 std::vector<std::shared_ptr<HybZono>> leaf_diffs; // init
422 for (auto& Z1_leaf : Z1_leaves)
423 {
424 leaf_diffs.push_back(pontry_diff(Z1_leaf, Z2_zono, false));
425 }
426 return union_of_many(leaf_diffs);
427 }
428 else if (Z2.is_hybzono())
429 {
430 // get leaves and convert to zonos
431 auto Z2_CZ_leaves = Z2.get_leaves();
432 std::vector<Zono> Z2_leaves;
433 for (auto& CZ : Z2_CZ_leaves)
434 {
435 Z2_leaves.push_back(*CZ.to_zono_approx());
436 }
437
438 // take intersection of pontry diffs
439 std::unique_ptr<HybZono> Z_out; // declare
440 for (auto& Z2_leaf : Z2_leaves)
441 {
442 if (!Z_out) {
443 Z_out = pontry_diff(Z1, Z2_leaf, false);
444 }
445 else {
446 Z_out = intersection(*Z_out, *pontry_diff(Z1, Z2_leaf, false));
447 }
448 }
449 return Z_out;
450 }
451 else
452 {
453 // convert to zono subtrahend (do nothing if already zono)
454 auto Z2_CZ = dynamic_cast<const ConZono*>(&Z2);
455 Zono Z2_zono = *Z2_CZ->to_zono_approx();
456
457 // get [G; A] matrices
458 const Eigen::SparseMatrix<zono_float> GA1 = vcat<zono_float>(Z1.G, Z1.A);
459 Eigen::SparseMatrix<zono_float> GA2 = Z2_zono.G;
460 GA2.conservativeResize(Z2_zono.n + Z1.nC, Z2_zono.nG);
461
462 Eigen::SparseMatrix<zono_float> M;
463 if (Z1.n + Z1.nC == Z1.nG)
464 {
465 // solve GA1^{-1}*GA2
466 Eigen::SparseLU<Eigen::SparseMatrix<zono_float>> lu(GA1);
467 if (lu.info() != Eigen::Success)
468 {
469 throw std::runtime_error("Pontryagin difference: failed to peform LU decomposition. Most likely [G; A] is not full row rank");
470 }
471 M = lu.solve(GA2);
472 }
473 else
474 {
475 // solve pinv(GA1)*GA2 where pinv is right pseudoinverse
476 Eigen::SimplicialLDLT<Eigen::SparseMatrix<zono_float>> ldlt(GA1*GA1.transpose());
477 if (ldlt.info() != Eigen::Success)
478 {
479 throw std::runtime_error("Pontryagin difference: failed to perform LDLT decomposition. Most likely [G; A] is not full row rank");
480 }
481 const Eigen::SparseMatrix<zono_float> ldlt_sol = ldlt.solve(GA2);
482 M = GA1.transpose()*ldlt_sol;
483 }
484
485 std::vector<Eigen::Triplet<zono_float>> triplets;
486 triplets.reserve(Z1.nG);
487 Eigen::Vector<zono_float, -1> e (Z1.nG);
488 for (int i=0; i < Z1.nG; ++i)
489 {
490 e.setZero();
491 e(i) = 1.0;
492 const zono_float d = 1-(e.transpose()*M).cwiseAbs().sum();
493 if (d < 0)
494 {
495 return std::make_unique<EmptySet>(Z1.n);
496 }
497 triplets.emplace_back(i, i, d);
498 }
499 Eigen::SparseMatrix<zono_float> D (Z1.nG, Z1.nG);
500 D.setFromSortedTriplets(triplets.begin(), triplets.end());
501
502 return std::make_unique<ConZono>(Z1.G*D, Z1.c - Z2_zono.c, Z1.A*D, Z1.b, false);
503 }
504}

506inline std::unique_ptr<HybZono> union_of_many(const std::vector<std::shared_ptr<HybZono>>& Zs_in, const bool preserve_sharpness, const bool expose_indicators)
507{
508 // remove empty sets from sets to be unioned
509 std::vector<std::shared_ptr<HybZono>> Zs;
510 for (auto& Z : Zs_in)
511 {
512 if (!Z->is_empty_set())
513 {
514 Zs.push_back(Z);
515 }
516 }
517
518 // check we are taking a union of at least one zonotope
519 if (Zs.empty())
520 {
521 throw std::invalid_argument("Union: empty input vector.");
522 }
523
524 // check dimensions
525 const int n = Zs[0]->n;
526 for (const auto & Z : Zs)
527 {
528 if (Z->n != n)
529 {
530 throw std::invalid_argument("Union: inconsistent dimensions.");
531 }
532 }
533
534 // make sure all Zs are using [0,1] generators
535 for (const auto & Z : Zs)
536 {
537 if (!Z->zero_one_form)
538 {
539 Z->convert_form();
540 }
541 }
542
543 // declare
544 std::vector<Eigen::Triplet<zono_float>> tripvec;
545 int rows = 0, cols = 0;
546 std::vector<int> idx_sum_to_1;
547 Eigen::SparseMatrix<zono_float> Gc, Gb, Ac, Ab;
548 Eigen::Vector<zono_float, -1> c, b;
549
550 if (preserve_sharpness)
551 {
552 // constraints
553
554 // Ac
555 tripvec.clear();
556 rows = 0;
557 cols = 0;
558 for (const auto & Z : Zs)
559 {
560 // equality constraints
561 get_triplets_offset<zono_float>(Z->Ac, tripvec, rows, cols);
562 rows += Z->nC;
563
564 // identity matrices
565 for (int i=0; i<Z->nGc; i++)
566 {
567 tripvec.emplace_back(rows+i, cols+i, one);
568 }
569 for (int i=0; i<Z->nG; i++)
570 {
571 tripvec.emplace_back(rows+i, cols+Z->nGc+i, one);
572 }
573
574 // increment
575 rows += Z->nG;
576 cols += Z->nGc + Z->nG;
577 }
578 Ac.resize(rows+1, cols); // last row all zeroes
579 Ac.setFromTriplets(tripvec.begin(), tripvec.end());
580
581 // Ab
582 tripvec.clear();
583 rows = 0;
584 cols = 0;
585 for (const auto & Z : Zs)
586 {
587 // equality constraints
588 get_triplets_offset<zono_float>(Z->Ab, tripvec, rows, cols);
589
590 // last column
591 for (int i=0; i<Z->nC; i++)
592 {
593 tripvec.emplace_back(rows+i, cols+Z->nGb, -Z->b(i));
594 }
595 rows += Z->nC;
596
597 // identity matrix
598 for (int i=0; i<Z->nGb; i++)
599 {
600 tripvec.emplace_back(rows+Z->nGc+i, cols+i, one);
601 }
602
603 // last column
604 for (int i=0; i<Z->nG; i++)
605 {
606 tripvec.emplace_back(rows+i, cols+Z->nGb, -one);
607 }
608
609 // increment
610 rows += Z->nG;
611 cols += Z->nGb + 1;
612
613 // track sum-to-1 binaries
614 idx_sum_to_1.push_back(cols-1);
615 }
616 // sum to 1 constraint
617 for (int & it : idx_sum_to_1)
618 {
619 tripvec.emplace_back(rows, it, one);
620 }
621 rows++;
622 Ab.resize(rows, cols);
623 Ab.setFromTriplets(tripvec.begin(), tripvec.end());
624
625 // b
626 b.resize(Ab.rows());
627 b.setZero();
628 b(Ab.rows()-1) = 1.0;
629
630 // generators
631 int n_out = n;
632 if (expose_indicators)
633 n_out += static_cast<int>(Zs.size());
634
635 // Gc
636 tripvec.clear();
637 cols = 0;
638 for (const auto & Z : Zs)
639 {
640 get_triplets_offset<zono_float>(Z->Gc, tripvec, 0, cols);
641 cols += 2*Z->nGc + Z->nGb;
642 }
643 Gc.resize(n_out, cols);
644 Gc.setFromTriplets(tripvec.begin(), tripvec.end());
645
646 // Gb
647 tripvec.clear();
648 cols = 0;
649 for (const auto & Z : Zs)
650 {
651 get_triplets_offset<zono_float>(Z->Gb, tripvec, 0, cols);
652 cols += Z->nGb;
653 for (int i=0; i<Z->n; i++)
654 {
655 tripvec.emplace_back(i, cols, Z->c(i));
656 }
657 cols += 1;
658 }
659 for (int i=n; i<n_out; ++i)
660 {
661 tripvec.emplace_back(i, idx_sum_to_1[i-n], one);
662 }
663 Gb.resize(n_out, cols);
664 Gb.setFromTriplets(tripvec.begin(), tripvec.end());
665
666 // c
667 c.resize(n_out);
668 c.setZero();
669 }
670 else
671 {
672 // constraints
673
674 // Ac
675 tripvec.clear();
676 rows = 0;
677 cols = 0;
678 for (const auto & Z : Zs)
679 {
680 // populate first row
681 for (int i=0; i<Z->nGc; i++)
682 {
683 tripvec.emplace_back(rows, cols+i, one);
684 }
685 tripvec.emplace_back(rows, cols+Z->nGc, static_cast<zono_float>(Z->nG));
686
687 // increment
688 rows++;
689
690 // equality constraints
691 get_triplets_offset<zono_float>(Z->Ac, tripvec, rows, cols);
692
693 // increment
694 rows += Z->nC;
695 cols += Z->nGc + 1;
696 }
697 Ac.resize(rows+1, cols); // last row all zeroes
698 Ac.setFromTriplets(tripvec.begin(), tripvec.end());
699
700 // Ab
701 tripvec.clear();
702 rows = 0;
703 cols = 0;
704 for (const auto & Z : Zs)
705 {
706 // populate first row
707 for (int i=0; i<Z->nGb; i++)
708 {
709 tripvec.emplace_back(rows, cols+i, one);
710 }
711 tripvec.emplace_back(rows, cols+Z->nGb, static_cast<zono_float>(-Z->nG));
712
713 // increment
714 rows++;
715
716 // equality constraints
717 get_triplets_offset<zono_float>(Z->Ab, tripvec, rows, cols);
718
719 // last column
720 for (int i=0; i<Z->nC; i++)
721 {
722 tripvec.emplace_back(rows+i, cols+Z->nGb, -Z->b(i));
723 }
724
725 // increment
726 rows += Z->nC;
727 cols += Z->nGb + 1;
728
729 // track sum-to-1 binaries
730 idx_sum_to_1.push_back(cols-1);
731 }
732 // sum to 1 constraint
733 for (int & it : idx_sum_to_1)
734 {
735 tripvec.emplace_back(rows, it, one);
736 }
737 rows++;
738 Ab.resize(rows, cols);
739 Ab.setFromTriplets(tripvec.begin(), tripvec.end());
740
741 // b
742 b.resize(Ab.rows());
743 b.setZero();
744 b(Ab.rows()-1) = 1.0;
745
746 // generators
747 int n_out = n;
748 if (expose_indicators)
749 n_out += static_cast<int>(Zs.size());
750
751 // Gc
752 tripvec.clear();
753 cols = 0;
754 for (const auto & Z : Zs)
755 {
756 get_triplets_offset<zono_float>(Z->Gc, tripvec, 0, cols);
757 cols += Z->nGc + 1;
758 }
759 Gc.resize(n_out, cols);
760 Gc.setFromTriplets(tripvec.begin(), tripvec.end());
761
762 // Gb
763 tripvec.clear();
764 cols = 0;
765 for (const auto & Z : Zs)
766 {
767 get_triplets_offset<zono_float>(Z->Gb, tripvec, 0, cols);
768 cols += Z->nGb;
769 for (int i=0; i<Z->n; i++)
770 {
771 tripvec.emplace_back(i, cols, Z->c(i));
772 }
773 cols += 1;
774 }
775 for (int i=n; i<n_out; ++i)
776 {
777 tripvec.emplace_back(i, idx_sum_to_1[i-n], one);
778 }
779 Gb.resize(n_out, cols);
780 Gb.setFromTriplets(tripvec.begin(), tripvec.end());
781
782 // c
783 c.resize(n_out);
784 c.setZero();
785 }
786
787 // check if known to be sharp
788 bool sharp = preserve_sharpness;
789 size_t i = 0;
790 while (sharp && i < Zs.size())
791 {
792 sharp = Zs[i]->sharp;
793 ++i;
794 }
795
796 // return hybrid zonotope
797 return std::make_unique<HybZono>(Gc, Gb, c, Ac, Ab, b, true, sharp);
798}

800inline std::unique_ptr<HybZono> cartesian_product(const HybZono& Z1, HybZono& Z2)
801{
802 // trivial case
803 if (Z1.is_empty_set() || Z2.is_empty_set()) {
804 return std::make_unique<EmptySet>(Z1.n + Z2.n);
805 }
806
807 // make sure Z1 and Z2 both using same generator range
808 if (Z1.zero_one_form != Z2.zero_one_form)
809 {
810 Z2.convert_form();
811 }
812
813 // declare
814 std::vector<Eigen::Triplet<zono_float>> tripvec;
815
816 // take Cartesian product
817 Eigen::SparseMatrix<zono_float> Gc (Z1.n + Z2.n, Z1.nGc + Z2.nGc);
818 get_triplets_offset<zono_float>(Z1.Gc, tripvec, 0, 0);
819 get_triplets_offset<zono_float>(Z2.Gc, tripvec, Z1.n, Z1.nGc);
820 Gc.setFromTriplets(tripvec.begin(), tripvec.end());
821
822 tripvec.clear();
823 Eigen::SparseMatrix<zono_float> Gb (Z1.n + Z2.n, Z1.nGb + Z2.nGb);
824 get_triplets_offset<zono_float>(Z1.Gb, tripvec, 0, 0);
825 get_triplets_offset<zono_float>(Z2.Gb, tripvec, Z1.n, Z1.nGb);
826 Gb.setFromTriplets(tripvec.begin(), tripvec.end());
827
828 Eigen::Vector<zono_float, -1> c (Z1.n + Z2.n);
829 c.segment(0, Z1.n) = Z1.c;
830 c.segment(Z1.n, Z2.n) = Z2.c;
831
832 tripvec.clear();
833 Eigen::SparseMatrix<zono_float> Ac (Z1.nC + Z2.nC, Z1.nGc + Z2.nGc);
834 get_triplets_offset<zono_float>(Z1.Ac, tripvec, 0, 0);
835 get_triplets_offset<zono_float>(Z2.Ac, tripvec, Z1.nC, Z1.nGc);
836 Ac.setFromTriplets(tripvec.begin(), tripvec.end());
837
838 tripvec.clear();
839 Eigen::SparseMatrix<zono_float> Ab (Z1.nC + Z2.nC, Z1.nGb + Z2.nGb);
840 get_triplets_offset<zono_float>(Z1.Ab, tripvec, 0, 0);
841 get_triplets_offset<zono_float>(Z2.Ab, tripvec, Z1.nC, Z1.nGb);
842 Ab.setFromTriplets(tripvec.begin(), tripvec.end());
843
844 Eigen::Vector<zono_float, -1> b (Z1.nC + Z2.nC);
845 b.segment(0, Z1.nC) = Z1.b;
846 b.segment(Z1.nC, Z2.nC) = Z2.b;
847
848 // return correct output type
849 if (Z1.is_hybzono() || Z2.is_hybzono())
850 return std::make_unique<HybZono>(Gc, Gb, c, Ac, Ab, b, Z1.zero_one_form);
851 else if (Z1.is_conzono() || Z2.is_conzono())
852 return std::make_unique<ConZono>(Gc, c, Ac, b, Z1.zero_one_form);
853 else if (Z1.is_zono() || Z2.is_zono())
854 return std::make_unique<Zono>(Gc, c, Z1.zero_one_form);
855 else
856 return std::make_unique<Point>(c);
857}

859inline std::unique_ptr<HybZono> constrain(HybZono& Z, const std::vector<Inequality> &ineqs, const Eigen::SparseMatrix<zono_float>& R)
860{
861 // trivial case
862 if (Z.is_empty_set()) {
863 return std::make_unique<EmptySet>(Z.n);
864 }
865
866 // handle default arguments
867 const Eigen::SparseMatrix<zono_float> * R_ptr = nullptr;
868 Eigen::SparseMatrix<zono_float> R_def;
869 if (R.rows() == 0 && R.cols() == 0)
870 {
871 R_def.resize(Z.n, Z.n);
872 R_def.setIdentity();
873 R_ptr = &R_def;
874 }
875 else
876 {
877 R_ptr = &R;
878 }
879
880 // check that dimensions match
881 for (const auto& ineq : ineqs)
882 {
883 if (R_ptr->rows() != ineq.get_n_dims())
884 throw std::invalid_argument("Inequality does not have the same number of dimensions as set");
885 }
886
887 // make sure Z in 0-1 form
888 if (!Z.is_0_1_form())
889 Z.convert_form();
890
891 // build constraints
892 std::vector<Eigen::Triplet<zono_float>> triplets_Ac, triplets_Ab;
893 Eigen::Vector<zono_float, -1> b_new (ineqs.size());
894 Eigen::Index n_cons = 0, n_slack = 0;
895 Eigen::SparseMatrix<zono_float, Eigen::RowMajor> RGc = (*R_ptr)*Z.Gc;
896 Eigen::SparseMatrix<zono_float, Eigen::RowMajor> RGb = (*R_ptr)*Z.Gb;
897 Eigen::Vector<zono_float, -1> Rc = (*R_ptr)*Z.c;
898
899 for (const auto& ineq : ineqs)
900 {
901 zono_float rhs;
902 switch (ineq.get_ineq_type())
903 {
904 case LESS:
905 rhs = ineq.get_rhs() - zono_eps;
906 break;
907 case GREATER:
908 rhs = ineq.get_rhs() + zono_eps;
909 break;
910 default:
911 rhs = ineq.get_rhs();
912 break;
913 }
914 zono_float gamma = rhs; // slack variable scaling
915 zono_float db = 0;
916 const auto ineq_type = ineq.get_ineq_type();
917
918 for (const auto& [idx, coeff] : ineq.get_terms())
919 {
920 const auto trips_Gc = get_triplets_row<zono_float>(RGc, idx);
921 for (const auto& trip : trips_Gc)
922 {
923 const zono_float val = coeff * trip.value();
924 triplets_Ac.emplace_back(static_cast<int>(n_cons), static_cast<int>(trip.col()), val);
925 if ((val < 0 && (ineq_type==LESS_OR_EQUAL || ineq_type==LESS)) || (val > 0 && (ineq_type==GREATER_OR_EQUAL || ineq_type==GREATER)))
926 gamma -= val;
927 }
928
929 const auto trips_Gb = get_triplets_row<zono_float>(RGb, idx);
930 for (const auto& trip : trips_Gb)
931 {
932 const zono_float val = coeff * trip.value();
933 triplets_Ab.emplace_back(static_cast<int>(n_cons), static_cast<int>(trip.col()), val);
934 if ((val < 0 && (ineq_type==LESS_OR_EQUAL || ineq_type==LESS)) || (val > 0 && (ineq_type==GREATER_OR_EQUAL || ineq_type==GREATER)))
935 gamma -= val;
936 }
937
938 const zono_float ddb = coeff * Rc(idx);
939 db -= ddb;
940 gamma -= ddb;
941 }
942
943 // rhs
944 b_new(n_cons) = rhs + db;
945
946 // add slack variable
947 if (ineq_type!=EQUAL)
948 {
949 triplets_Ac.emplace_back(static_cast<int>(n_cons), Z.nGc + static_cast<int>(n_slack), gamma);
950 ++n_slack; // increment slack variable index
951 }
952
953 // increment
954 ++n_cons;
955 }
956
957 Eigen::SparseMatrix<zono_float> Ac_cons (static_cast<Eigen::Index>(ineqs.size()), Z.nGc + n_slack);
958 Eigen::SparseMatrix<zono_float> Ab_cons (static_cast<Eigen::Index>(ineqs.size()), Z.nGb);
959 Ac_cons.setFromTriplets(triplets_Ac.begin(), triplets_Ac.end());
960 Ab_cons.setFromTriplets(triplets_Ab.begin(), triplets_Ab.end());
961
962 // set matrices / vectors
963 Eigen::SparseMatrix<zono_float> Z_Ac = Z.Ac;
964 Z_Ac.conservativeResize(Z.nC, Z.nGc + n_slack);
965 Eigen::SparseMatrix<zono_float> Ac = vcat(Z_Ac, Ac_cons);
966 Eigen::SparseMatrix<zono_float> Ab = vcat(Z.Ab, Ab_cons);
967 Eigen::Vector<zono_float, -1> b (Z.nC + static_cast<Eigen::Index>(ineqs.size()));
968 b.segment(0, Z.nC) = Z.b;
969 b.segment(Z.nC, static_cast<Eigen::Index>(ineqs.size())) = b_new;
970
971 Eigen::SparseMatrix<zono_float> Z_Gc = Z.Gc;
972 Z_Gc.conservativeResize(Z.n, Z.nGc + n_slack);
973
974 // output correct type
975 if (Z.is_hybzono())
976 return std::make_unique<HybZono>(Z_Gc, Z.Gb, Z.c, Ac, Ab, b, true);
977 else
978 return std::make_unique<ConZono>(Z_Gc, Z.c, Ac, b, true);
979}

981inline std::unique_ptr<HybZono> HybZono::do_complement(const zono_float delta_m, const bool remove_redundancy, const OptSettings &settings,
982 OptSolution* solution, const int n_leaves, const int contractor_iter)
983{
984 // make sure set in [-1,1] form
985 if (this->is_0_1_form()) this->convert_form();
986
987 // need to get leaves and do complement for each leaf if Z is a hybzono
988 auto leaves = this->get_leaves(remove_redundancy, settings, solution, n_leaves, contractor_iter);
989 if (leaves.empty())
990 {
991 throw std::runtime_error("HybZono complement: set is empty.");
992 }
993 std::vector<std::unique_ptr<HybZono>> complements; // init
994 for (auto& leaf : leaves)
995 {
996 complements.emplace_back(leaf.complement(delta_m));
997 }
998 std::unique_ptr<HybZono> Z_out;
999 for (auto& comp : complements)
1000 {
1001 if (!Z_out)
1002 {
1003 Z_out.reset(comp->clone());
1004 }
1005 else
1006 {
1007 Z_out = intersection(*Z_out, *comp);
1008 }
1009 }
1010 return Z_out;
1011}

1013inline std::unique_ptr<HybZono> ConZono::do_complement(const zono_float delta_m, bool, const OptSettings&, OptSolution*, int, int)
1014{
1015 // make sure in [-1,1] form
1016 if (this->is_0_1_form()) this->convert_form();
1017
1018 // get a value lambda_m such that lambda_m >= max{ ||lambda||_infty : |[G^T A^T] lambda| <= 1 }
1019
1020 // construct the matrix [G^T A^T]
1021 const auto GT = this->G.transpose();
1022 const auto AT = this->A.transpose();
1023 const Eigen::SparseMatrix<zono_float, Eigen::RowMajor> GTAT = hcat<zono_float>(GT, AT); // convert to row-major
1024
1025 // get the smallest non-zero element in the matrix, making sure that all rows have at least one non-zero element
1026 zono_float min_non_zero = std::numeric_limits<zono_float>::max();
1027 for (int row=0; row<GTAT.outerSize(); ++row)
1028 {
1029 bool value_exists = false;
1030 for (Eigen::SparseMatrix<zono_float, Eigen::RowMajor>::InnerIterator it(GTAT, row); it; ++it)
1031 {
1032 if (it.value() < min_non_zero && std::abs(it.value()) > zono_eps)
1033 {
1034 min_non_zero = it.value();
1035 }
1036 value_exists = true;
1037 }
1038 if (!value_exists)
1039 {
1040 std::stringstream ss;
1041 ss << "ConZono complement: row " << row << " of [G^T A^T] has no non-zero elements.";
1042 throw std::runtime_error(ss.str());
1043 }
1044 }
1045
1046 // value for lambda_m
1047 const zono_float lambda_m = 1 / min_non_zero;
1048
1049 // m value
1050 const zono_float m = delta_m + 1;
1051
1052 // interval sets

1749inline void ConZono::constraint_reduction()
1750{
1751 // make sure there are constraints to remove
1752 if (this->nC == 0) return;
1753
1754 // put set into [-1, 1] form
1755 if (this->zero_one_form) this->convert_form();
1756
1757 // execute algorithm 1 from paper
1758 Eigen::Vector<zono_float, -1> x_lb (this->nG);
1759 Eigen::Vector<zono_float, -1> x_ub (this->nG);
1760 x_lb.setConstant(-1);
1761 x_ub.setConstant(1);
1762 Box E (x_lb, x_ub);
1763 x_lb.setConstant(-std::numeric_limits<zono_float>::infinity());
1764 x_ub.setConstant(std::numeric_limits<zono_float>::infinity());
1765 Box R (x_lb, x_ub);
1766 Eigen::SparseMatrix<zono_float, Eigen::RowMajor> A_rm = this->A;
1767 for (int i=0; i<this->nC; ++i)
1768 {
1769 for (Eigen::SparseMatrix<zono_float, Eigen::RowMajor>::InnerIterator it_j(A_rm, i); it_j; ++it_j)
1770 {
1771 const zono_float a_ij = it_j.value();
1772 Interval y (this->b(i) / a_ij, this->b(i) / a_ij);
1773 for (Eigen::SparseMatrix<zono_float, Eigen::RowMajor>::InnerIterator it_k(A_rm, i); it_k; ++it_k)
1774 {
1775 if (it_j.col() == it_k.col()) continue;
1776 y = y - E[it_k.col()].to_interval()*(it_k.value()/a_ij);
1777 }
1778 R[it_j.col()].intersect_assign(R[it_j.col()], y.as_view());
1779 E[it_j.col()].intersect_assign(E[it_j.col()], R[it_j.col()]);
1780 }
1781 }
1782
1783 // make sure conzono isn't empty (interval check)
1784 for (int j=0; j<this->nG; ++j)
1785 {
1786 if (E[j].is_empty())
1787 throw std::runtime_error("ConZono constraint reduction: set is empty");
1788 }
1789
1790 // build Q matrix
1791 Eigen::SparseMatrix<zono_float> Q (this->nG + this->nC, this->nG + this->nC);
1792
1793 Eigen::SparseMatrix<zono_float> I_nG (this->nG, this->nG);
1794 I_nG.setIdentity();
1795 const Eigen::SparseMatrix<zono_float> Phi = this->G.transpose()*this->G + I_nG;
1796
1797 std::vector<Eigen::Triplet<zono_float>> triplets;
1798 get_triplets_offset<zono_float>(Phi, triplets, 0, 0);
1799 get_triplets_offset<zono_float>(this->A, triplets, this->nG, 0);
1800 get_triplets_offset<zono_float>(this->A.transpose(), triplets, 0, this->nG);
1801 Q.setFromTriplets(triplets.begin(), triplets.end());
1802
1803 // factorize Q
1804 Eigen::SimplicialLDLT<Eigen::SparseMatrix<zono_float>> Q_ldlt (Q);
1805 if (Q_ldlt.info() != Eigen::Success)
1806 throw std::runtime_error("ConZono constraint reduction: Q matrix factorization failed, most likely A is not full row rank.");
1807
1808 // get estimated Hausdorff error for eliminating each generator
1809 std::vector<std::pair<int, zono_float>> haus_vec; // (index, error)
1810 haus_vec.reserve(this->nG);
1811 auto shift_permute = [size=this->nG + this->nC + 1](const int start_index, const int end_index) -> Eigen::PermutationMatrix<Eigen::Dynamic, Eigen::Dynamic>
1812 {
1813 assert(start_index >= 0 && end_index >= 0 && start_index < size && end_index < size && start_index < end_index);
1814 Eigen::PermutationMatrix<Eigen::Dynamic, Eigen::Dynamic> P(size);
1815 for (int i=0; i<start_index; ++i)
1816 {
1817 P.indices()[i] = i;
1818 }
1819 P.indices()[start_index] = end_index;
1820 for (int i=start_index+1; i<=end_index; ++i)
1821 {
1822 P.indices()[i] = i-1;
1823 }
1824 for (int i=end_index+1; i<size; ++i)
1825 {
1826 P.indices()[i] = i;
1827 }
1828 return P;
1829 };
1830 Eigen::Vector<zono_float, -1> e_j (this->nG + this->nC);
1831 for (int j=0; j<this->nG; ++j)
1832 {
1833 // get r_j
1834 const zono_float r_j = std::max<zono_float>(zero, std::max<zono_float>(std::abs(R[j].to_interval().lb), std::abs(R[j].to_interval().ub)) - one);
1835 if (r_j < zono_eps)
1836 {
1837 haus_vec.emplace_back(j, zero);
1838 continue;
1839 }
1840
1841 // linear system matrix
1842 e_j.setZero();
1843 e_j(j) = 1;
1844 const Eigen::Vector<zono_float, -1> Qinv_e_j = Q_ldlt.solve(e_j);
1845
1846 triplets.clear();
1847 for (int i=0; i<this->nG + this->nC; ++i)
1848 {
1849 triplets.emplace_back(i, i, one);
1850 if (i == j)
1851 {
1852 triplets.emplace_back(this->nG + this->nC, j, one); // extra row for e_j^T
1853 }
1854 }
1855 for (int i=0; i<this->nG + this->nC; ++i)
1856 {
1857 triplets.emplace_back(i, this->nG + this->nC, Qinv_e_j(i));
1858 }
1859 Eigen::SparseMatrix<zono_float> M (this->nG + this->nC + 1, this->nG + this->nC + 1);
1860 M.setFromSortedTriplets(triplets.begin(), triplets.end());
1861
1862 // RHS for linear system
1863 Eigen::Vector<zono_float, -1> rhs (this->nG + this->nC + 1);
1864 rhs.setZero();
1865 rhs(this->nG + this->nC) = r_j;
1866
1867 // permutation matrices to make system upper triangular
1868 const auto P_R = shift_permute(j, this->nG + this->nC);
1869 const auto P_L = shift_permute(j, this->nG + this->nC - 1);
1870
1871 // solve linear system in permuted space
1872 const Eigen::SparseMatrix<zono_float> M_perm = P_L*M*P_R.inverse();
1873 const Eigen::Vector<zono_float, -1> rhs_perm = P_L*rhs;
1874 const Eigen::Vector<zono_float, -1> y_perm = M_perm.triangularView<Eigen::Upper>().solve(rhs_perm);
1875 const Eigen::Vector<zono_float, -1> y = P_R*y_perm;
1876
1877 // get Hausdorff distance estimate
1878 const Eigen::Vector<zono_float, -1> d = y.segment(0, this->nG);
1879 const zono_float haus_j = (this->G*d).norm() + d.norm();
1880 haus_vec.emplace_back(j, haus_j);
1881 }
1882
1883 // sort by Hausdorff error
1884 std::sort(haus_vec.begin(), haus_vec.end(), [](const auto &a, const auto &b) { return a.second < b.second; });
1885
1886 Eigen::SparseMatrix<zono_float> Ea (this->nG, this->nC); // init
1887 int gen_remove = -1;
1888 int cons_remove = -1;
1889 for (const auto& [j, err] : haus_vec)
1890 {
1891 // loop through column k of A to find a constraint to remove
1892 for (Eigen::SparseMatrix<zono_float>::InnerIterator it(this->A, j); it; ++it)
1893 {
1894 if (std::abs(it.value()) > zono_eps)
1895 {
1896 triplets.emplace_back(j, static_cast<int>(it.row()), one/it.value());
1897 Ea.insert(j, it.row()) = 1/it.value();
1898 gen_remove = j;
1899 cons_remove = static_cast<int>(it.row());
1900 break;
1901 }
1902 }
1903
1904 // check if done
1905 if (gen_remove != -1 && cons_remove != -1)
1906 break;
1907 }
1908 if (gen_remove == -1 || cons_remove == -1)
1909 throw std::runtime_error("ConZono: constraint reduction cannot find valid constraint to remove");
1910
1911 // apply algorithm from Scott paper
1912 const Eigen::SparseMatrix<zono_float> Lambda_G = G*Ea;
1913 const Eigen::SparseMatrix<zono_float> Lambda_A = A*Ea;
1914 Eigen::SparseMatrix<zono_float> Gp = this->G - Lambda_G*this->A;
1915 Eigen::Vector<zono_float, -1> cp = this->c + Lambda_G*this->b;
1916 Eigen::SparseMatrix<zono_float> Ap = this->A - Lambda_A*this->A;
1917 Eigen::Vector<zono_float, -1> bp = this->b - Lambda_A*this->b;
1918
1919 // generator removal matrix
1920 triplets.clear();
1921 for (int j=0; j<gen_remove; ++j)
1922 {
1923 triplets.emplace_back(j, j, one);
1924 }
1925 for (int j=gen_remove+1; j<this->nG; ++j)
1926 {
1927 triplets.emplace_back(j, j-1, one);
1928 }
1929 Eigen::SparseMatrix<zono_float> dG (this->nG, this->nG - 1);
1930 dG.setFromSortedTriplets(triplets.begin(), triplets.end());
1931
1932 // constraint removal matrix
1933 triplets.clear();
1934 for (int i=0; i<cons_remove; ++i)
1935 {
1936 triplets.emplace_back(i, i, one);
1937 }
1938 for (int i=cons_remove+1; i<this->nC; ++i)
1939 {
1940 triplets.emplace_back(i-1, i, one);
1941 }
1942 Eigen::SparseMatrix<zono_float> dA (this->nC - 1, this->nC);
1943 dA.setFromSortedTriplets(triplets.begin(), triplets.end());
1944
1945 // update
1946 this->set(Gp*dG, cp, dA*Ap*dG, dA*bp, false);
1947}

1949inline std::unique_ptr<Zono> ConZono::to_zono_approx() const
1950{
1951 // check for case that there are no constraints
1952 if (this->nG == 0)
1953 {
1954 return std::make_unique<Point>(this->c);
1955 }
1956 if (this->nC == 0)
1957 {
1958 return std::make_unique<Zono>(this->G, this->c, this->zero_one_form);
1959 }
1960
1961 // compute SVD of A
1962 const Eigen::BDCSVD<Eigen::Matrix<zono_float, -1, -1>, Eigen::ComputeFullV | Eigen::ComputeFullU> svd (this->A.toDense());
1963 const Eigen::Vector<zono_float, -1>& sin_vals = svd.singularValues();
1964
1965 const int n = this->nG;
1966 const int r = static_cast<int>(svd.rank());
1967 const int d = n-r;
1968 const Eigen::Matrix<zono_float, -1, -1> Vr = svd.matrixV().block(0, 0, n, r);
1969 const Eigen::Matrix<zono_float, -1, -1> Vd = svd.matrixV().block(0, r, n, d);
1970
1971 // get xi_tilde_r
1972 const Eigen::Vector<zono_float, -1> b_tilde = svd.matrixU().transpose()*this->b;
1973 const Eigen::Vector<zono_float, -1> xi_tilde_r = b_tilde.segment(0, r).array() / sin_vals.segment(0, r).array();
1974
1975 // build bounding zonotope
1976 const Eigen::SparseMatrix<zono_float> Vd_VdT_sp = (Vd*Vd.transpose()).sparseView();
1977 const Eigen::SparseMatrix<zono_float> G_zono = this->G*Vd_VdT_sp;
1978 const Eigen::Vector<zono_float, -1> c_zono = this->c + this->G*(Vr*xi_tilde_r);
1979 return std::make_unique<Zono>(G_zono, c_zono, this->zero_one_form);
1980}
